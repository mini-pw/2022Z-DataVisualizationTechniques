library(stringr)
library(ggplot2)

#edytowanie danych z pliku csv
df <- read.csv(file = '\\Users\\Admin\\Downloads\\EUR.csv')
df <- str_split_fixed(df$Data.EUR..Cena.kupna..EUR..Cena.sprzedazy..EURO, ";", 3)
df <- as.data.frame(df)
df[,2] <- as.numeric(df[,2]) 
df[,3] <- as.numeric(df[,3]) 
df[,1] <- as.POSIXct(strptime(df[,1], format ="%Y-%m-%d %H:%M:%S"))
colnames(df) <- c("data", "sprzeda", "kupno")

#tworzenie wykresu
colors <- c("kupno" = "green", "sprzeda" = "red")
ggplot(df, aes(x = data)) + 
  geom_line(aes(y = sprzeda, color = "sprzeda"), size = 0.4) +
  geom_line(aes(y = kupno, color = "kupno"), size = 0.4) +
  labs(title = "Warto euro w przeliczeniu na PLN",
       subtitle = "13.03.2020-29.10.2021", 
       y = "Cena euro w PLN",
       x = "Data",
       color = "Legenda") +
    scale_color_manual(values = colors)+
  theme(legend.position = "top")
